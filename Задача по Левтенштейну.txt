#include <iostream>
#include <string>
#include <vector>
using namespace std;

int MIN(int a, int b, int c) 
{
	if (a <= b && a <= c)
		return a;
	if (b <= c && b <= a)
		return b;
	else
		return c;
}

int main() 
{
	int InsertPrice = 0, DeletePrice = 0, ResPrice = 0;
	string first, second;
	getline(cin, first);
	getline(cin, second);
	cin >> InsertPrice >> DeletePrice >> ResPrice;
	int n = int(first.length()), m = int(second.length());
	vector<vector<int>> matrix(n + 1, vector<int>(m + 1));
	matrix[0][0] = 0;
	for (int j = 1; j <= m; ++j)
		matrix[0][j] = matrix[0][j - 1] + InsertPrice;
	for (int i = 1; i <= n; ++i) 
	{
		matrix[i][0] = matrix[i - 1][0] + DeletePrice;
		for (int j = 1; j <= m; ++j) 
		{
			matrix[i][j] = MIN(
			(matrix[i - 1][j] + DeletePrice),
			(matrix[i][j - 1] + InsertPrice),
			(matrix[i - 1][j - 1] + ResPrice)
			);
		}
	}
	cout << matrix[n][m];
	return 0;
}
